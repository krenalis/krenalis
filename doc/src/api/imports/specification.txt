
{%%
var ResourceByID = func() map[string]map[string]any {
    var spec map[string]any
    data := render "/api/spec.json"
    err := unmarshalJSON(string(data), &spec)
    if err != nil {
        panic(err)
    }
    resourceByID := map[string]map[string]any{}
    for resource in spec.resources.([]any) {
        r := resource.(map[string]any)
        resourceByID[r.id.(string)] = r
    }
    return resourceByID
}()
%%}
